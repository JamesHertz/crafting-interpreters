program   => statement* "EOF" ;

statement => printStmt | exprStmt;

printStmt =>  "print" expression ";" ;
exprStmt  =>  expression ";" ;

statement  => equality ;
equality   => comparison ( ( "==" | "!=" ) equality )* ;
comparison => term ( ( ">"| ">=" | "<"| "<=" ) comparison )* ;
term   => factor ( ( "+" | "-" ) term )* ;
factor => unary ( ( "*" | "/" ) factor )* ;

unary   => primary | ( "-" | "!" ) unary ;
primary => "(" expression ")" | NUMBER | STRING ;
